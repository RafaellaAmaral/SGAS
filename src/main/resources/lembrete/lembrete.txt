#### Postgres via DOCKER

# Opção 1) Execução no host
docker run --name my-postgres --network host -e POSTGRES_USER=postgres -e POSTGRES_PASSWORD=postgres -e POSTGRES_DB=SGAS -v pg_data:/var/lib/postgresql  postgres:latest
docker run --name my-pgadmin  --network host -e PGADMIN_DEFAULT_EMAIL=root@gmail.com -e PGADMIN_DEFAULT_PASSWORD=postgres dpage/pgadmin4

# Opão 2) Criar uma rede e interconectar
docker network create --driver bridge my-network
docker network ls
docker run --name my-postgres --network my-network -p 5432:5432 -e POSTGRES_USER=postgres -e POSTGRES_PASSWORD=postgres -e POSTGRES_DB=SGAS -v pg_data:/var/lib/postgresql  postgres:latest
docker run --name my-pgadmin --network my-network -p 15432:80 -e PGADMIN_DEFAULT_EMAIL=root@gmail.com -e PGADMIN_DEFAULT_PASSWORD=postgres dpage/pgadmin4
